    def emissions(url):
    # Kirjoita vastauksesi tähän alle:
    import pandas as pd
    df = pd.read_csv(url,
    engine='python',
    encoding='utf-8-sig',
    sep='\\t',
    usecols=['country_code','main activity sector name', 'value', 'year"'],
    converters={'year"': (lambda x : x[:-1])}
    )
    df.columns = df.columns.str.replace('"', '')
    df.columns = df.columns.str.replace('main activity sector name', 'sector')
    df.columns = df.columns.str.replace('value', 'emissions')
    df = df[df['sector'] != '20-99 All stationary installations']
    df = df[df['sector'] != '21-99 All industrial installations (excl. combustion)']
    df2 = df[df['year'].isin(['2010', '2011', '2012', '2013', '2014', '2015'])]
    df3 = pd.DataFrame(df2.groupby(df2['year'])['emissions'].sum())
    df3['change'] = df3['emissions'].pct_change()*100
    df3['change'] = df3['change'].round(1)
    df3['cumsum'] = df3['emissions'].cumsum()
    tulos = df3
